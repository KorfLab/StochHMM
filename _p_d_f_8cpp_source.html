81  //!Poisson Binomial Probability mass function
82  //!Iterative calculations don't make this appropriate for using with emissions
83  //! \param k Value or Number of successful trials
84  //! \param p Probability of Success for each trial of N trials
85  double poisson_binomial_pdf(size_t k, std::vector<double>& p){
86  if (k > p.size()){
87  std::cerr << "Poisson Binomial PDF Function: k > n \n";
88  exit(2);
89  }
90 
91  //Calculate K=0
92  double prob(1.0);
93  for(size_t i=0; i < p.size(); ++i){
94  prob*=(1-p[i]);
95  }
96 
97 
98  if(k > 0){
99 
100  //Calculate T(i)
101  std::vector<double> t(0,p.size());
102  for(size_t i=0; i < p.size(); ++i){
103  for (size_t j = 0; j < p.size(); ++j){
104  t[i] += pow(p[j]/(1-p[j]), i);
105  }
106  }
107 
108  double temp(0);
109  for(size_t i=0; i < k; ++i){
110  temp+= pow(-1,i-1) * prob * t[i];
111  prob = temp;
112  }
113 
114  prob *= 1/k;
115  }
116 
117  return prob;
118  }
